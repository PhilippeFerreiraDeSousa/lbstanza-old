Sample proj matching proposal
conantest01/stanza.proj:
########
package my-stz-package requires :
  ccfiles :
    "zip-wrapper.o"

file "zip-wrapper.o" requires :
  foreign-package(conan) :
    "goodzip/2.11.3"

foreign-package-params(conan) :
  downloaded-package-dirs: "myconanpackages"
  output-include-file: "build/buildinfoargs.txt"

compile file "zip-wrapper.o" :
  "g++ {.}/my-wrapper.cpp -o {.}/zip-wrapper.o {CONAN-INCLUDES}"

########
